public class BallContext {    private static Map<String, TaiQiu> taiqiuLst = new HashMap<String, TaiQiu>();
    public static void addBallLst(TaiQiu taiQiu) {        taiqiuLst.put(taiQiu.getBallNumber(), taiQiu);    }
    public static Map<String, TaiQiu> getAllBalls() {        return taiqiuLst;    }
    public static TaiQiu getBall(String BallNumber) {        return taiqiuLst.get(BallNumber);    }
    public static double angleCalculate(double subtractX, double subtractY) {        double angle = 0.0;        if (subtractY == 0) {            if (subtractX > 0) {                angle = 0;            } else {                angle = 180;            }        } else if (subtractX == 0) {            if (subtractY > 0) {                angle = 90;            } else {                angle = 270;            }        } else {            angle = Math.round(Math.atan(subtractX / subtractY) / Math.PI * 180);            if (subtractX > 0 && subtractY < 0) {                angle = 270 - angle;            } else if (subtractX < 0 && subtractY > 0) {                angle = 90 - angle;            } else if (subtractX < 0 && subtractY < 0) {                angle = 180 + angle;            }        }        return angle;    }}

import java.util.ArrayList;import java.util.List;import java.util.Map.Entry;
public class BallIntersect {
    public static List<String> intersect(double centerx, double centerY, String ballNumber) {        double radius = 15.0;// 定义圆的半径        List<String> ballNumbers = new ArrayList<>();        TaiQiu ball = BallContext.getBall(ballNumber);        double angle = ball.getAngle();        for (Entry<String, TaiQiu> entry : BallContext.getAllBalls().entrySet()) {            TaiQiu intersectBall = entry.getValue();            if (ballNumber.equals(intersectBall.getBallNumber())) {                continue;            }            double s = 0;            double xAbs = Math.abs(centerx - intersectBall.getCenterX());            double yAbs = Math.abs(centerY - intersectBall.getCenterY());
            s = Math.round(Math.sqrt(xAbs * xAbs + yAbs * yAbs));            if (xAbs == 0) {                s = yAbs;            }            if (yAbs == 0) {                s = yAbs;            }            // 相交判断            if (s <= 2 * radius) {                ballNumbers.add(entry.getValue().getBallNumber());
                double pointAngle = BallContext.angleCalculate(centerx - intersectBall.getCenterX(),                        centerY - intersectBall.getCenterY());                double halfAngle = 0.0;                if (pointAngle > 180) {                    halfAngle = pointAngle - 180 - angle;                } else {                    halfAngle = pointAngle + 180 - angle;                }                intersectBall.setAngle(pointAngle + halfAngle);                if ((pointAngle + halfAngle) > 180) {                    ball.setAngle(pointAngle + halfAngle - 180);                } else {                    ball.setAngle(180 - pointAngle - halfAngle);                }            }        }        return ballNumbers;    }}

        Point2D center2d = new Point2D.Double();        double edge1CenterX = parentX;        double edge1CenterY = parentY;        if (angle <= 90) {            edge1CenterX = edge1CenterX + rotate * cos;            edge1CenterY = edge1CenterY + rotate * sin;        } else if (angle <= 180) {            edge1CenterX = edge1CenterX - rotate * cos;            edge1CenterY = edge1CenterY + rotate * sin;        } else if (angle <= 270) {            edge1CenterX = edge1CenterX - rotate * cos;            edge1CenterY = edge1CenterY - rotate * sin;        } else if (angle <= 360) {            edge1CenterX = edge1CenterX + rotate * cos;            edge1CenterY = edge1CenterY - rotate * sin;        }        center2d.setLocation(edge1CenterX, edge1CenterY);        return center2d;    }        private void edgePaint(Graphics2D g2d,Ellipse2D shape,Point2D edge) {        Area edgeShape = new Area(shape);        Ellipse2D edgeCircle = new Ellipse2D.Double();        edgeCircle.setFrameFromCenter(edge, new Point2D.Double(edge.getX() - 10, edge.getY() - 10));        edgeShape.intersect(new Area(edgeCircle));        g2d.fill(edgeShape);    }    }
